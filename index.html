let foods = []; // 🌟 核心：从 GitHub 获取数据 // 替换下面的 [用户名] 和 [仓库名] const dataUrl = 'https://raw.githubusercontent.com/[你的用户名]/[你的仓库名]/main/data.json'; async function loadDatabase() { try { // 先读那 50 个真实的 const response = await fetch(dataUrl); const realFoods = await response.json(); // 然后在内存里自动变出 950 个假的补齐 let fakeFoods = []; for(let i = 51; i <= 1000; i++) { fakeFoods.push({ "name": `金宝第 ${i} 号隐藏美食`, "tip": `这是系统为您生成的第 ${i} 条神秘贴士，去探索一下吧！` }); } // 把真的和假的合并 (Array Concatenation) foods = realFoods.concat(fakeFoods); document.getElementById('dbStatus').innerText = `✅ 已同步全量数据库 (50真 + 950假)`; } catch (error) { console.error("加载失败:", error); } } // 页面加载时自动运行 loadDatabase(); function startRolling() { if (foods.length === 0) return alert("数据库还在加载中..."); const nameEl = document.getElementById('foodName'); const tipEl = document.getElementById('foodTip'); const btn = document.getElementById('pickBtn'); btn.disabled = true; let count = 0; const timer = setInterval(() => { const randomTemp = foods[Math.floor(Math.random() * foods.length)]; nameEl.innerText = randomTemp.name; count++; if (count > 15) { clearInterval(timer); const final = foods[Math.floor(Math.random() * foods.length)]; nameEl.innerText = final.name; tipEl.innerText = "💡 " + final.tip; btn.disabled = false; } }, 80); }
