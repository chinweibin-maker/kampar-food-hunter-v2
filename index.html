let foods = []; // 🌟 1. 记得把这里的 [用户名] 和 [仓库名] 换成你真实的 GitHub 信息 const dataUrl = 'https://raw.githubusercontent.com/chinweibin-maker/kampar-food-hunter-v2/refs/heads/main/data.json'; async function loadDatabase() { try { // 🌟 2. 抓取你在 GitHub 写的 50 条真实美食 const response = await fetch(dataUrl); const realFoods = await response.json(); // 🌟 3. 核心：在内存中自动变出剩下的 950 条数据 (Data Augmentation) // 这样你就不需要手动在 data.json 里写一千行了 let fakeFoods = []; for (let i = 51; i <= 1000; i++) { fakeFoods.push({ "name": `金宝隐藏美食 #${i}`, "tip": `这是系统为您发掘的第 ${i} 处神秘美食，快去金宝街头偶遇吧！` }); } // 🌟 4. 合并数据：真 50 + 假 950 = 总共 1000 foods = realFoods.concat(fakeFoods); document.getElementById('dbStatus').innerText = `✅ 已同步全量数据库 (${foods.length} 家店)`; } catch (error) { console.error("加载失败:", error); document.getElementById('dbStatus').innerText = "❌ 数据库同步失败，请检查 URL 或网络"; } } // 页面加载时自动运行 loadDatabase(); function startRolling() { if (foods.length === 0) return alert("数据库还在加载中..."); const nameEl = document.getElementById('foodName'); const tipEl = document.getElementById('foodTip'); const btn = document.getElementById('pickBtn'); btn.disabled = true; let count = 0; // 🌟 5. 抽奖动画：在 1000 条数据里快速滚动 const timer = setInterval(() => { const randomTemp = foods[Math.floor(Math.random() * foods.length)]; nameEl.innerText = randomTemp.name; count++; if (count > 20) { // 稍微增加一点滚动次数，让 1000 条看起来更震撼 clearInterval(timer); const final = foods[Math.floor(Math.random() * foods.length)]; nameEl.innerText = final.name; tipEl.innerText = "💡 " + final.tip; btn.disabled = false; } }, 60); // 滚动速度加快，更有抽奖感 }
