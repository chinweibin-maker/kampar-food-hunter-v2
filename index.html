let foods = []; // 🌟 1. 记得把这里的 [用户名] 和 [仓库名] 换成你真实的 GitHub 信息 const dataUrl = 'https://raw.githubusercontent.com/chinweibin-maker/kampar-food-hunter-v2/main/data.json'; async function loadDatabase() { try { console.log("正在尝试连接数据库..."); const response = await fetch(dataUrl); if (!response.ok) { throw new Error(`服务器响应错误: ${response.status}`); } const realFoods = await response.json(); // 自动补齐逻辑... let fakeFoods = []; for (let i = realFoods.length + 1; i <= 1000; i++) { fakeFoods.push({ "name": `金宝隐藏美食 #${i}`, "tip": `探索第 ${i} 号神秘美食！` }); } foods = realFoods.concat(fakeFoods); document.getElementById('dbStatus').innerText = `✅ 数据库就绪 (${foods.length} 家店)`; console.log("数据库加载成功！"); } catch (error) { console.error("加载失败原因:", error); document.getElementById('dbStatus').innerText = `❌ 同步失败: ${error.message}`; } } // 页面加载时自动运行 loadDatabase(); function startRolling() { if (foods.length === 0) return alert("数据库还在加载中..."); const nameEl = document.getElementById('foodName'); const tipEl = document.getElementById('foodTip'); const btn = document.getElementById('pickBtn'); btn.disabled = true; let count = 0; // 🌟 5. 抽奖动画：在 1000 条数据里快速滚动 const timer = setInterval(() => { const randomTemp = foods[Math.floor(Math.random() * foods.length)]; nameEl.innerText = randomTemp.name; count++; if (count > 20) { // 稍微增加一点滚动次数，让 1000 条看起来更震撼 clearInterval(timer); const final = foods[Math.floor(Math.random() * foods.length)]; nameEl.innerText = final.name; tipEl.innerText = "💡 " + final.tip; btn.disabled = false; } }, 60); // 滚动速度加快，更有抽奖感 }
