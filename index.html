let foods = []; // 🌟 核心：从 GitHub 获取数据 // 替换下面的 [用户名] 和 [仓库名] const dataUrl = 'https://raw.githubusercontent.com/[你的用户名]/[你的仓库名]/main/data.json'; async function loadDatabase() { try { const response = await fetch(dataUrl); foods = await response.json(); document.getElementById('dbStatus').innerText = `✅ 已同步 GitHub 数据库 (${foods.length} 家店)`; } catch (error) { console.error("加载失败:", error); document.getElementById('dbStatus').innerText = "❌ 数据库同步失败，请检查网络"; } } // 页面加载时自动运行 loadDatabase(); function startRolling() { if (foods.length === 0) return alert("数据库还在加载中..."); const nameEl = document.getElementById('foodName'); const tipEl = document.getElementById('foodTip'); const btn = document.getElementById('pickBtn'); btn.disabled = true; let count = 0; const timer = setInterval(() => { const randomTemp = foods[Math.floor(Math.random() * foods.length)]; nameEl.innerText = randomTemp.name; count++; if (count > 15) { clearInterval(timer); const final = foods[Math.floor(Math.random() * foods.length)]; nameEl.innerText = final.name; tipEl.innerText = "💡 " + final.tip; btn.disabled = false; } }, 80); }
